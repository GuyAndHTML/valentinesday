const yesButton = document.getElementById('btn-yes');
const noButton = document.getElementById('btn-no');
const messageDisplay = document.getElementById('message');

// What happens when you click "Yes"
yesButton.addEventListener('click', () => {
    messageDisplay.innerHTML = "Yay! You've made me the happiest person ever! ðŸ’–ðŸŽ‰";
    // Optional: Add confetti or play music here later!
});

// The magic: "No" button runs away on hover
noButton.addEventListener('mouseenter', () => {
    // Get the container's dimensions to keep the button inside
    const container = document.querySelector('.container');
    const containerRect = container.getBoundingClientRect();
    const noButtonRect = noButton.getBoundingClientRect();

    // Calculate max positions
    const maxX = containerRect.width - noButtonRect.width;
    const maxY = containerRect.height - noButtonRect.height;

    // Generate random new positions
    const randomX = Math.floor(Math.random() * maxX);
    const randomY = Math.floor(Math.random() * maxY);

    // Apply the new position (relative to the container)
    noButton.style.position = 'relative';
    noButton.style.left = `${randomX}px`;
    noButton.style.top = `${randomY}px`;
});

// Optional: If they manage to click the moving "No" button
noButton.addEventListener('click', () => {
    messageDisplay.innerHTML = "Nice try! But the 'Yes' button looks nicer, doesn't it? ðŸ˜‰";
});
